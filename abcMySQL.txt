(

   NAME   VARCHAR(100)
);

CREATE TABLE ISSUES
(
   ID  SERIAL PRIMARY KEY,
   ISSUESLIST_ID  BIGINT UNSIGNED NOT NULL,
   SUMMARY    VARCHAR(100),
   DESCRIPTION  VARCHAR(1000),
   FOREIGN KEY (ISSUESLIST_ID) REFERENCES ISSUESLISTS (ID)
);
