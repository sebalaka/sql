SELECT U.NAME, I.SUMMARY
FROM ISSUESLISTS U
JOIN ISSUES I ON U.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY AS TASKS_LIST, U.FIRSTNAME AS FIRSTNAME, U.LASTNAME AS LASTNAME
FROM ISSUES I
JOIN USERS U ON I.ISSUESLIST_ID = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, 
COUNT(*) AS ISSUES_LEFT
FROM ISSUES I JOIN USERS U ON U.ID = I.ISSUESLIST_ID
GROUP BY U.ID;
